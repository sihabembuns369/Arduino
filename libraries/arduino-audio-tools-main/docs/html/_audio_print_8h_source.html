504  void flushMixer() {
505  LOGD("flush");
506  bool result = false;
507 
508  // determine ringbuffer with mininum available data
509  size_t samples = size_bytes/sizeof(T);
510  for (int j=0;j<output_count;j++){
511  samples = MIN(samples, (size_t) buffers[j]->available());
512  }
513 
514  if (samples>0){
515  result = true;
516  // mix data from ringbuffers to output
517  output.resize(samples);
518  memset(output.data(),0, samples*sizeof(T));
519  for (int j=0;j<output_count;j++) {
520  float weight = weights[j];
521  // sum up input samples to result samples
522  for (int i=0;i<samples; i++){
523  output[i] += weight * buffers[j]->read() / total_weights;
524  }
525  }
526 
527  // write output
528  LOGD("write to final out: %d", samples*sizeof(T));
529  p_final_output->write((uint8_t*)output.data(), samples*sizeof(T));
530  }
531  stream_idx = 0;
532  return;
533  }
