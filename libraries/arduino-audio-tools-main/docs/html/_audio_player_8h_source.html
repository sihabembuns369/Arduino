340  if (active) {
341  TRACED();
342  if (delay_if_full!=0 && p_final_print!=nullptr && p_final_print->availableForWrite()==0){
343  // not ready to do anything - so we wait a bit
344  delay(delay_if_full);

346  }
347  // handle sound
348  if (copier.copy() || timeout == 0) {
349  // reset timeout
350  timeout = millis() + p_source->timeoutAutoNext();
351  }
352  // move to next stream after timeout
353  moveToNextFileOnTimeout();
354  } else {

356  // only single copy
