185  // prevent npe

201  while (idx<frames){

204  if (p_out->availableForWrite()<sizeof(T)){

208  }

210  }
211  // save last samples
212  setupLastSamples(data, frames-1);

218  inline T getValue(T *data,float frame_idx, int channel){
219  // provide value if number w/o digits
220  if (frame_idx==(int)frame_idx){
221  return lookup(data, frame_idx-1, channel);
222  }
223  // interpolate value
224  int frame_idx1 = frame_idx;
225  int frame_idx0 = frame_idx1-1;
226  T val0 = lookup(data, frame_idx0, channel);
227  T val1 = lookup(data, frame_idx1, channel);
228  T diff = val1 - val0;
229  if (diff==0){
230  // No difference
231  return val0;
232  }
233  // interpolate value
234  float delta = (frame_idx - frame_idx0) - 1;
235  T diffEffective = diff * delta;
236  T result = val0+diffEffective;
237 
238 
239  return result;
240  }
