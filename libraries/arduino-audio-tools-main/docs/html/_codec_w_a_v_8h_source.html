60  while (!eof()) {
61  uint32_t tag, tag2, length;
62  tag = read_tag();
63  if (eof())
64  break;
65  length = read_int32();
66  if (!length || length >= 0x7fff0000) {
67  headerInfo.is_streamed = true;
68  length = ~0;
69  }
70  if (tag != TAG('R', 'I', 'F', 'F') || length < 4) {
71  seek(length, SEEK_CUR);
72  continue;
73  }
74  tag2 = read_tag();
75  length -= 4;
76  if (tag2 != TAG('W', 'A', 'V', 'E')) {
77  seek(length, SEEK_CUR);
78  continue;
79  }
80  // RIFF chunk found, iterate through it
81  while (length >= 8) {
82  uint32_t subtag, sublength;
83  subtag = read_tag();
84  if (eof())
85  break;
86  sublength = read_int32();
87  length -= 8;
88  if (length < sublength)
89  break;
90  if (subtag == TAG('f', 'm', 't', ' ')) {
91  if (sublength < 16) {
92  // Insufficient data for 'fmt '
93  break;
94  }
95  headerInfo.format = read_int16();
96  headerInfo.channels = read_int16();
97  headerInfo.sample_rate = read_int32();
98  headerInfo.byte_rate = read_int32();
99  headerInfo.block_align = read_int16();
100  headerInfo.bits_per_sample = read_int16();
101  if (headerInfo.format == 0xfffe) {
102  if (sublength < 28) {
103  // Insufficient data for waveformatex
104  break;
105  }
106  skip(8);
107  headerInfo.format = read_int32();
108  skip(sublength - 28);
109  } else {
110  skip(sublength - 16);
111  }
112  headerInfo.is_valid = true;
113  } else if (subtag == TAG('d', 'a', 't', 'a')) {
114  sound_pos = tell();
115  headerInfo.data_length = sublength;
116  if (!headerInfo.data_length || headerInfo.is_streamed) {
117  headerInfo.is_streamed = true;
118  logInfo();
119  return;
120  }
121  seek(sublength, SEEK_CUR);
122  } else {
123  skip(sublength);
124  }
125  length -= sublength;
126  }
127  if (length > 0) {
128  // Bad chunk?
129  seek(length, SEEK_CUR);
130  }
131  }
