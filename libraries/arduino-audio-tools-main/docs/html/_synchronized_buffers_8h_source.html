163 class SynchronizedBuffer : public BaseBuffer<T> {
164 public:
165  SynchronizedBuffer(BaseBuffer<T> &buffer, Mutex &mutex) {
166  p_buffer = &buffer;
167  p_mutex = &mutex;
168  }
169 
170  // reads a single value
171  T read() override {
172  TRACED();
173  LockGuard guard(p_mutex);
174  return p_buffer->read();
175  }
176 
177  // reads multiple values
178  int readArray(T data[], int len) {
179  TRACED();
180  LockGuard guard(p_mutex);
181  int lenResult = MIN(len, available());
182  for (int j = 0; j < lenResult; j++) {
183  data[j] = p_buffer->read();
184  }
185  return lenResult;
186  }
187 
188  int writeArray(const T data[], int len) {
189  LOGD("%s: %d", LOG_METHOD, len);
190  LockGuard guard(p_mutex);
191  int result = 0;
192  for (int j = 0; j < len; j++) {
193  if (p_buffer->write(data[j]) == 0) {
194  break;
195  }
196  result = j + 1;
197  }
198  return result;
199  }
200 
201  // peeks the actual entry from the buffer
202  T peek() override {
203  TRACED();
204  LockGuard guard(p_mutex);
205  return p_buffer->peek();
206  }
207 
208  // checks if the buffer is full
209  bool isFull() override { return p_buffer->isFull(); }
210 
211  bool isEmpty() { return available() == 0; }
212 
213  // write add an entry to the buffer
214  bool write(T data) override {
215  TRACED();
216  LockGuard guard(p_mutex);
217  return p_buffer->write(data);
218  }
219 
220  // clears the buffer
221  void reset() override {
222  TRACED();
223  LockGuard guard(p_mutex);
224  p_buffer->reset();
225  }
226 
227  // provides the number of entries that are available to read
228  int available() override {
229  TRACED();
230  return p_buffer->available();
231  }
232 
233  // provides the number of entries that are available to write
234  int availableForWrite() override {
