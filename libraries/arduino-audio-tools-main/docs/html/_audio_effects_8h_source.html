255  int frames = length / sizeof(T) / info.channels;
256  size_t result_size = 0;
257 
258  if (p_io->available()<(sizeof(T)*info.channels)){
259  return 0;
260  }
261 
262  for (int count=0;count<frames;count++){
263  // determine sample by combining all channels in frame
264  T result_sample = 0;
265  for (int ch=0;ch<info.channels;ch++){
266  T sample;
267  p_io->readBytes((uint8_t*)&sample, sizeof(T));
268  result_sample += sample / info.channels;
269  }

273  result_sample = effects[j]->process(result_sample);
274  }

276  // write result multiplying channels
277  T* p_buffer = ((T*)buffer)+(count*info.channels);
278  for (int ch=0;ch<info.channels;ch++){
279  p_buffer[ch] = result_sample;
